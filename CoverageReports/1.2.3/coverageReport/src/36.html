using System;
using System.Collections.Generic;
using System.Reflection;
using SolutionInspector.Commons.Extensions;
using SolutionInspector.Configuration.Attributes;

namespace SolutionInspector.Configuration.Validation.Static.Validators
{
  internal class NoSubelementsBesidesDynamicDefaultCollectionValidator : StaticConfigurationValidatorBase
  {
    private PropertyInfo _dynamicDefaultCollectionProperty;
    private readonly List<PropertyInfo> _otherSubelementProperties = new List<PropertyInfo>();

    public override void BeginTypeValidation (Type configurationElementType, ReportValidationError reportValidationError)
    {
      _dynamicDefaultCollectionProperty = null;
      _otherSubelementProperties.Clear();
    }

    public override void EndTypeValidation (Type configurationElementType, ReportValidationError reportValidationError)
    {
      if (_dynamicDefaultCollectionProperty == null || _otherSubelementProperties.Count == 0)
        return;

      var otherProperties = _otherSubelementProperties.ConvertAndJoin(p => $"'{p.Name}'", ", ");

      reportValidationError(
          _dynamicDefaultCollectionProperty,
          "When a configuration element contains a dynamic configuration collection that is a default collection no other configuration properties that " +
          $"require a nested element are allowed (the following properties required nested elements: {otherProperties}). Either remove the aforementioned" +
          "properties or change the dynamic configuration collection to no longer be a default collection.");
    }

    public override void ValidateCollection (PropertyInfo property, ConfigurationCollectionAttribute attribute, ReportValidationError reportValidationError)
    {
      _otherSubelementProperties.Add(property);
    }

    public override void ValidateSubelement (PropertyInfo property, ConfigurationSubelementAttribute attribute, ReportValidationError reportValidationError)
    {
      _otherSubelementProperties.Add(property);
    }

    public override void ValidateDynamicCollection (PropertyInfo property, DynamicConfigurationCollectionAttribute attribute, ReportValidationError reportValidationError)
    {
      if (_dynamicDefaultCollectionProperty == null && attribute.IsDefaultCollection)
        _dynamicDefaultCollectionProperty = property;
      else
        _otherSubelementProperties.Add(property);
    }
  }
}
